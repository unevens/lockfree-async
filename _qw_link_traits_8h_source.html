32 /*
33  QwLinkTraits is an adapter that linked list and node-based queue classes
34  use to access the "next link" pointer (or sometimes, "previous link" pointer)
35  stored in each node.
36 
37  QwLinkTraits is parameterised by Node type and link index. An instantiation
38  of QwLinkTraits thus provides access to a single link field of a particular
39  node type using QwLinkTraits::load() and QwLinkTraits::store() functions.
40  If the link supports atomic operations, QwLinkTraits::is_atomic == true
41  and QwLinkTraits also provides atomic_load() and atomic_store() functions.
42 
43  The default implementation of QwLinkTraits requires that the Node type has
44  a publicly accessible array of link pointers named `links_`.
45  The zero-based LINK_INDEX template parameter determines the element of
46  the `links_` array that is operated upon.
47 
48  The intent is that clients will declare an enum indicating the role(s)
49  of each link. e.g.:
50 
51  struct Node {
52 
53  enum LinkIndices {
54  ACTOR_MESSAGE_QUEUE_LINK = 0,
55  CLIENT_WHEN_NOT_SENT_LINK_INDEX = 0,
56  SERVER_INTERNAL_QUEUEING_LINK_INDEX = 0,
57  REQUEST_CHAINING_LINK_WHEN_SENT = 1,
58  CLIENT_EXCLUSIVE_LINK_INDEX = 2,
59  LINK_COUNT = 3
60  };
61 
62  Node *links_[LINK_COUNT];
63 
64  ...
65  };
66 
67  This link array representation makes it easy to declare Nodes that provide
68  multiple links for simultaneous use by multiple lists and queues. Perhaps
69  more importantly, it also allows the same link to be "overloaded" for
70  multiple uses, and when this is done, it is clear from the enum values that
71  the index is being overloaded (see link index 0 in the example above).
72 
73  Clients can specialize QwLinkTraits to support alternative link
74  representations, or to customize link access.
75 
76  The QwLinkTraits template defined in this file is the default/base-case
77  template. The base-case template delegates to either QwDefaultAtomicLinkTraitsImpl or
78  QwDefaultNonAtomicLinkTraitsImpl depending on whether or not the Node's links_ are
79  declared as std::atomic.
80 */

87 template<typename node_type, int LINK_INDEX>
88 inline static std::size_t offsetof_link()
89 {
90 #if __cplusplus >= 201103L // C++11 version.
91  // The following is well defined according to discussion and links here:
92  // https://stackoverflow.com/questions/49775980/is-it-well-defined-to-use-stdaddressof-on-non-active-union-members
93  // I prefer this to offsetof() because offsetof() gives warnings in C++11 for non-standard-layout types.
94  // Arguably should use offsetof() in C++17 and later where it is conditionally supported for other types.
95  union U {
96  char c;
97  node_type n;
98  constexpr U() : c(0) {}
99  };
100  static const U u; // REVIEW: do we get better compiler output by making this non-static?
101  return static_cast<std::size_t>(
102  reinterpret_cast<const char*>(std::addressof(u.n.links_[LINK_INDEX])) -
103  reinterpret_cast<const char*>(&u));
104  // notes:
105  // - the union layout rules guarantee that &u == &u.n (use &u because we know operator& isn't defined)
106  // - use std::addressof to avoid problems with pointer-like objects that define operator&
107 #else
108  return offsetof(node_type, links_[LINK_INDEX]);
109 #endif
110 }

114 template<typename NodePtrT, int LINK_INDEX>
115 struct QwDefaultNonAtomicLinkTraitsImpl {
116  typedef typename std::remove_pointer<NodePtrT>::type node_type;
117  typedef NodePtrT node_ptr_type;
118  typedef const node_type* const_node_ptr_type;
119 
120  static constexpr bool is_atomic = false;
121 
122  static node_ptr_type load(const_node_ptr_type n)
123  {
124  return static_cast<node_ptr_type>(n->links_[ LINK_INDEX ]); // downcast to node_ptr_type in case links_[n] is a ptr to base class type
125  }
126 
127  static void store(node_ptr_type n, node_ptr_type x) // n->link = x
128  {
129  n->links_[ LINK_INDEX ] = x;
130  }
131 
132  static std::size_t offsetof_link()
133  {
134  return Qw::impl::offsetof_link<node_type, LINK_INDEX>();
135  }
136 };

140 template<typename NodePtrT, int LINK_INDEX>
141 struct QwDefaultAtomicLinkTraitsImpl {
142  typedef typename std::remove_pointer<NodePtrT>::type node_type;
143  typedef NodePtrT node_ptr_type;
144  typedef const node_type* const_node_ptr_type;
145 
146  // atomic accessors
147 
148  // is_atomic=true indicates that the link can be loaded and stored atomically
149  // using the atomic_load() and atomic_store() functions.
150  // Containers that require atomic link access check this tag at compile time.
151  static constexpr bool is_atomic = true;
152 
153  static node_ptr_type atomic_load(const_node_ptr_type n, std::memory_order order) // downcast to node_ptr_type in case links_[n] is a ptr to base class type
154  {
155  return static_cast<node_ptr_type>(n->links_[LINK_INDEX].load(order));
156  }
157 
158  static void atomic_store(node_ptr_type n, node_ptr_type x, std::memory_order order) // n->link = x
159  {
160  n->links_[LINK_INDEX].store(x, order);
161  }
162 
163  // non-atomic accessors
164 
165  // IMPORTANT: if is_atomic == true, then all link accesses must be atomic,
166  // not just atomic_load() and atomic_store() (this is because simultaneous
167  // atomic and non-atomic access violates strict aliasing rules.)
168  // Use memory_order_relaxed for the non-atomic accessors, since these will
169  // be invoked by single-threaded containers.
170 
171  static node_ptr_type load(const_node_ptr_type n)
172  {
173  return atomic_load(n, std::memory_order_relaxed);
174  }
175 
176  static void store(node_ptr_type n, node_ptr_type x) // n->link = x
177  {
178  atomic_store(n, x, std::memory_order_relaxed);
179  }
180 
181  static std::size_t offsetof_link()
182  {
183  return Qw::impl::offsetof_link<node_type, LINK_INDEX>();
184  }
185 };
