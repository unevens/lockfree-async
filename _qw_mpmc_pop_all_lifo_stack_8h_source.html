34 /*
35  QwMpmcPopAllLifoStack is a lock-free concurrent LIFO stack that
36  provides push() and pop_all() operations. No single-node pop()
37  operation is provided.
38 
39  All operations may be invoked concurrently.
40 
41  Implemented using the "IBM Freelist" LIFO algorithm.
42 
43  The algorithm doesn't need ABA protection because it does not provide a pop() operation.
44  pop_all() is not subject to the ABA problem because it swaps in a nullptr value and never
45  requires comparison to a non-nullptr value.
46  See ALGORITHMS.txt
47 */
48 
49 template<typename NodePtrT, int NEXT_LINK_INDEX>
50 class QwMpmcPopAllLifoStack{
51  typedef QwLinkTraits<NodePtrT, NEXT_LINK_INDEX> nextlink;
52  // Note: there is no requirement for nextlink to be atomic, since
53  // it is never accessed from multiple threads simultaneously, and
54  // transmission between threads is always mediated by an atomic
55  // compare exchange with appropriate memory barrier.
56 
57 public:
58  typedef typename nextlink::node_type node_type;
59  typedef typename nextlink::node_ptr_type node_ptr_type;
60  typedef typename nextlink::const_node_ptr_type const_node_ptr_type;
61 
62 private:
63  std::atomic<node_ptr_type> top_;
64 
65 #if (QW_VALIDATE_NODE_LINKS == 1)
66  void CHECK_NODE_IS_UNLINKED(const_node_ptr_type n) const
67  {
68 #ifndef NDEBUG
69  assert(nextlink::load(n) == nullptr); // (require unlinked)
70  // Node could be unlinked (nullptr next ptr) but still at top of stack; check that:
71  assert(n != top_.load(std::memory_order_relaxed));
72  // Note: we can't check that the node is not referenced by some other list
73 #else
74  if (!(nextlink::load(n) == nullptr)) { std::abort(); } // (require unlinked)
75  if (!(n != top_.load(std::memory_order_relaxed))) { std::abort(); }
76 #endif
77  }
78 
79  void CLEAR_NODE_LINKS_FOR_VALIDATION(node_ptr_type n) const
80  {
81  nextlink::store(n, nullptr);
82  }
83 #else
84  void CHECK_NODE_IS_UNLINKED(const_node_ptr_type) const {}
85  void CLEAR_NODE_LINKS_FOR_VALIDATION(node_ptr_type) const {}
86 #endif
87 
88 public:
89  QwMpmcPopAllLifoStack()
90  : top_(nullptr)
91  {}
92 
93  // The following four push variants use the same algorithm. Theh differ
94  // only in whether one or multiple items are pushed, and whether they
95  // provide the was-empty check.
96  // push(n), push(n, &wasEmpty), push_multiple(a, b) and push_multiple(a, b, &wasEmpty)
97 
98  void push(node_ptr_type node)
99  {
100  CHECK_NODE_IS_UNLINKED(node);
101 
102  node_ptr_type top = top_.load(std::memory_order_relaxed);
103  do {
104  nextlink::store(node, top);
105  // A fence is needed here for two reasons:
106  // 1. so that node's payload gets written before node becomes visible to consumer
107  // 2. ensure that node->next <-- top is written before top <-- node
108  } while (top_.compare_exchange_strong(top, node,
109  /*success:*/ std::memory_order_release,
110  /*failure:*/ std::memory_order_relaxed) == false);
111  }
112 
113  void push(node_ptr_type node, bool& wasEmpty)
114  {
115  CHECK_NODE_IS_UNLINKED(node);
116 
117  node_ptr_type top = top_.load(std::memory_order_relaxed);
118  do {
119  nextlink::store(node, top);
120  // A fence is needed here for two reasons:
121  // 1. so that node's payload gets written before node becomes visible to consumer
122  // 2. ensure that node->next <-- top is written before top <-- node
123  } while (top_.compare_exchange_strong(top, node,
124  /*success:*/ std::memory_order_release,
125  /*failure:*/ std::memory_order_relaxed) == false);
126 
127  wasEmpty = (top==nullptr);
128  }
129 
130  // push linked list link from front through to back
131  void push_multiple(node_ptr_type front, node_ptr_type back)
132  {
133  CHECK_NODE_IS_UNLINKED(back);
134 
135  node_ptr_type top = top_.load(std::memory_order_relaxed);
136  do {
137  nextlink::store(back, top);
138  // A fence is needed here for two reasons:
139  // 1. so that node's payload gets written before node becomes visible to consumer
140  // 2. ensure that back->next <-- top is written before top <-- front
141  } while (top_.compare_exchange_strong(top, front,
142  /*success:*/ std::memory_order_release,
143  /*failure:*/ std::memory_order_relaxed) == false);
144  }
145 
146  void push_multiple(node_ptr_type front, node_ptr_type back, bool& wasEmpty)
147  {
148  CHECK_NODE_IS_UNLINKED(back);
149 
150  node_ptr_type top = top_.load(std::memory_order_relaxed);
151  do {
152  nextlink::store(back, top);
153  // A fence is needed here for two reasons:
154  // 1. so that node's payload gets written before node becomes visible to consumer
155  // 2. ensure that back->next <-- top is written before top <-- front
156  } while (top_.compare_exchange_strong(top, front,
157  /*success:*/ std::memory_order_release,
158  /*failure:*/ std::memory_order_relaxed) == false);
159 
160  wasEmpty = (top==nullptr);
161  }
162 
163  bool empty() const
164  {
165  return (top_.load(std::memory_order_relaxed) == nullptr);
166  }
167 
168  node_ptr_type pop_all()
169  {
170  // acquire fence for all captured node data
171  return top_.exchange(nullptr, std::memory_order_acquire); // we'll return the first item
172  }
173 };
