138 template<typename T, class Action>
139 inline void
140 handleMessageStack(MessageNode<T>* head, Action action)
141 {
142  if (!head) {
143  return;
144  }
145  // gotta handle the message in FIFO order, so we cache that order using the
146  // MessageNode<T>::prev() link poiner
147  head->prev() = nullptr;
148  while (head->next()) {
149  head->next()->prev() = head;
150  head = head->next();
151  }
152  // and we loop from the new head (which is the last node of the lifo stack)
153  // using the prev pointer
154  while (head) {
155  action(head);
156  head = head->prev();
157  }
158 }

165 template<typename T>
166 inline void
167 freeMessageStack(MessageNode<T>* head)
168 {
169  while (head) {
170  auto next = head->next();
171  delete head;
172  head = next;
173  }
174 }

190 template<typename T = char>
191 class Messenger final
192 {
193  LifoStack<T> lifo;
194  LifoStack<T> storage;
195 
196 public:
201  void send(MessageNode<T>* node) { lifo.push(node); }
202 
208  void send(T message)
209  {
210  auto node = storage.pop_all();
211  bool fromStorage = true;
212  if (!node) {
213  node = new MessageNode<T>(std::move(message));
214  fromStorage = false;
215  }
216  if (fromStorage) {
217  node->set(std::move(message));
218  auto next = node->next();
219  node->next() = nullptr;
220  if (next) {
221  storage.push_multiple(next, next->last());
222  }
223  }
224  lifo.push(node);
225  }
226 
236  bool receiveLastMessage(T& message)
237  {
238  MessageNode<T>* head = lifo.pop_all();
239  if (!head) {
240  return false;
241  }
242  message = std::move(head->get());
243  storage.push_multiple(head, head->last());
244  return true;
245  }

295  void preallocateNodes(
296  int numNodesToPreallocate,
297  std::function<T(void)> initializer = [] { return T(); })
298  {
299  MessageNode<T>* head = nullptr;
300  MessageNode<T>* it = nullptr;
301  for (int i = 0; i < numNodesToPreallocate; ++i) {
302  auto node = new MessageNode<T>(initializer());
303  if (it) {
304  it->next() = node;
305  it = node;
306  }
307  else if (!head) {
308  head = it = node;
309  }
310  }
311  recycle(head);
312  }
313 
314  ~Messenger()
315  {
316  freeMessageStack(lifo.pop_all());
317  freeMessageStack(storage.pop_all());
318  }
319 };

329 template<typename T, class Action>
330 inline int
331 receiveAndHandleMessageStack(Messenger<T>& messenger, Action action)
332 {
333  auto messages = messenger.receiveAllNodes();
334  auto numMessages = messages->count();
335  handleMessageStack(messages, action);
336  messenger.recycle(messages);

345 template<typename T>
346 class MessageBuffer
347 {
348  LifoStack<T> storage;
349  int desiredBufferSize;
350  int minSize;
351  std::atomic<int> actualSize;
352  std::function<T(void)> initializer;
353 
354 public:
361  MessageBuffer(
362  int desiredBufferSize,
363  int minSize,
364  std::function<T(void)> initializer = [] { return T(); })
365  : desiredBufferSize{ desiredBufferSize }
366  , actualSize{ 0 }
367  , minSize{ minSize }
368  , initializer{ initializer }
369  {
370  assert(desiredBufferSize >= minSize);
371  replenish();
372  }
373 
374  ~MessageBuffer() { freeMessageStack(storage.pop_all()); }
375 
379  void replenish()
380  {
381  auto head = storage.pop_all();
382  int numNodes = head ? head->count() : 0;
383  auto prev = head ? head->last() : nullptr;
384  for (int i = numNodes; i < desiredBufferSize; ++i) {
385  auto node = new MessageNode<T>(initializer());
386  if (prev) {
387  prev->next() = node;
388  }
389  if (!head) {
390  head = node;
391  }
392  prev = node;
393  }
394  storage.push_multiple(head, head->last());
395  actualSize = desiredBufferSize;
396  }
397 
407  MessageNode<T>* getMessageNode(bool onlyIfAvailable = false,
408  bool* emptyBufferFlag = nullptr)
409  {
410  auto* head = storage.pop_all();
411  if (!head) {
412  if (emptyBufferFlag) {
413  *emptyBufferFlag = true;
414  }
415  if (onlyIfAvailable) {
416  return nullptr;
417  }
418  return new MessageNode<T>(initializer());
419  }
420  if (emptyBufferFlag) {
421  *emptyBufferFlag = false;
422  }
423  storage.push_multiple(head->next(), head->last());
424  --actualSize;
425  head->next() = nullptr;
426  return head;
427  }
428 
432  int getNumAvailableNodes() const { return actualSize; }
433 
438  void maintenance()
439  {
440  if (actualSize < minSize) {
441  replenish();
442  }
443  }
444 };
